    <script>

        async function fetchItalics() {

            const url = document.getElementById('urlInput').value;

            const italicList = document.getElementById('italicList');

            italicList.innerHTML = ''; // Clear previous results


            try {

                const response = await fetch(url);

                if (!response.ok) {

                    throw new Error(`HTTP error! status: ${response.status}`);

                }

                const text = await response.text();

                const parser = new DOMParser();

                const doc = parser.parseFromString(text, 'text/html');

                const italics = doc.querySelectorAll('i, em');


                italics.forEach(item => {

                    const li = document.createElement('li');

                    li.textContent = item.textContent.trim();

                    italicList.appendChild(li);

                });


                if (italics.length === 0) {

                    const li = document.createElement('li');

                    li.textContent = 'No italicized words found.';

                    italicList.appendChild(li);

                }

            } catch (error) {

                console.error('Error fetching the page:', error);

                const li = document.createElement('li');

                li.textContent = `Error fetching the page: ${error.message}`;

                italicList.appendChild(li);

            }

        }

    </script>
